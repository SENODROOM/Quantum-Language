// 2. Mini Math Expression Interpreter (Lexer → Parser → Evaluator)
// Supports: +  -  *  /  ^  ()  variables  let x = expr

function tokenize(expr) {
  const tokens = [];
  let i = 0;
  while (i < expr.length) {
    if (/\s/.test(expr[i])) { i++; continue; }
    if (/\d/.test(expr[i])) {
      let num = "";
      while (i < expr.length && /[\d.]/.test(expr[i])) num += expr[i++];
      tokens.push({ type: "NUM", val: parseFloat(num) });
    } else if (/[a-z]/i.test(expr[i])) {
      let word = "";
      while (i < expr.length && /[a-z]/i.test(expr[i])) word += expr[i++];
      tokens.push({ type: word === "let" ? "LET" : "VAR", val: word });
    } else {
      tokens.push({ type: "OP", val: expr[i++] });
    }
  }
  return tokens;
}

function parse(tokens) {
  let pos = 0;
  const peek = () => tokens[pos];
  const eat = (v) => { const t = tokens[pos++]; if (v && t?.val !== v) throw `Expected ${v}`; return t; };

  function primary() {
    const t = peek();
    if (t?.type === "NUM") { pos++; return { type: "num", val: t.val }; }
    if (t?.type === "VAR") { pos++; return { type: "var", name: t.val }; }
    if (t?.val === "(") { eat("("); const e = addSub(); eat(")"); return e; }
    throw `Unexpected: ${t?.val}`;
  }

  function power() {
    let left = primary();
    while (peek()?.val === "^") { eat("^"); left = { type: "^", left, right: primary() }; }
    return left;
  }

  function mulDiv() {
    let left = power();
    while (["*", "/"].includes(peek()?.val)) {
      const op = eat().val; left = { type: op, left, right: power() };
    }
    return left;
  }

  function addSub() {
    let left = mulDiv();
    while (["+", "-"].includes(peek()?.val)) {
      const op = eat().val; left = { type: op, left, right: mulDiv() };
    }
    return left;
  }

  if (tokens[0]?.type === "LET") {
    eat(); const name = eat().val; eat("="); return { type: "let", name, expr: addSub() };
  }
  return addSub();
}

function evaluate(node, env = {}) {
  if (node.type === "num") return node.val;
  if (node.type === "var") {
    if (!(node.name in env)) throw `Undefined variable: ${node.name}`;
    return env[node.name];
  }
  if (node.type === "let") { env[node.name] = evaluate(node.expr, env); return env[node.name]; }
  const l = evaluate(node.left, env), r = evaluate(node.right, env);
  return { "+": l+r, "-": l-r, "*": l*r, "/": l/r, "^": l**r }[node.type];
}

// 5. A* Pathfinding Visualizer
const GRID_W = 40, GRID_H = 18;

function createGrid(wallDensity = 0.28) {
  return Array.from({ length: GRID_H }, (_, r) =>
    Array.from({ length: GRID_W }, (_, c) =>
      (r === 0 && c === 0) || (r === GRID_H-1 && c === GRID_W-1)
        ? 0 : (Math.random() < wallDensity ? 1 : 0)
    )
  );
}

function heuristic(a, b) {
  return Math.abs(a.r - b.r) + Math.abs(a.c - b.c);
}

function aStar(grid, start, end) {
  const key = (r, c) => `${r},${c}`;
  const open = [{ ...start, g: 0, f: heuristic(start, end), parent: null }];
  const closed = new Set();
  const best = { [key(start.r, start.c)]: 0 };

  while (open.length) {
    open.sort((a, b) => a.f - b.f);
    const cur = open.shift();
    const k = key(cur.r, cur.c);
    if (closed.has(k)) continue;
    closed.add(k);

    if (cur.r === end.r && cur.c === end.c) {
      const path = [];
      let node = cur;
      while (node) { path.unshift({ r: node.r, c: node.c }); node = node.parent; }
      return { path, visited: closed.size };
    }

    for (const [dr, dc] of [[-1,0],[1,0],[0,-1],[0,1]]) {
      const nr = cur.r + dr, nc = cur.c + dc;
      if (nr < 0 || nr >= GRID_H || nc < 0 || nc >= GRID_W) continue;
      if (grid[nr][nc] === 1) continue;
      const nk = key(nr, nc);
      if (closed.has(nk)) continue;
      const g = cur.g + 1;
      if (best[nk] !== undefined && best[nk] <= g) continue;
      best[nk] = g;
      open.push({ r: nr, c: nc, g, f: g + heuristic({ r: nr, c: nc }, end), parent: cur });
    }
  }
  return null;
}

function render(grid, path, visited) {
  const pathSet = new Set(path?.map(p => `${p.r},${p.c}`) ?? []);
  const rows = grid.map((row, r) =>
    row.map((cell, c) => {
      const k = `${r},${c}`;
      if (r === 0 && c === 0) return "S";
      if (r === GRID_H-1 && c === GRID_W-1) return "E";
      if (pathSet.has(k)) return "█";
      return cell === 1 ? "▓" : "·";
    }).join("")
  );
  return rows.join("\n");
}

// 2. Markov Chain Text Generator
function buildChain(text, order = 2) {
  const chain = {};
  const words = text.split(/\s+/);
  for (let i = 0; i < words.length - order; i++) {
    const key = words.slice(i, i + order).join(" ");
    const next = words[i + order];
    if (!chain[key]) chain[key] = [];
    chain[key].push(next);
  }
  return chain;
}

function generate(chain, order = 2, length = 80) {
  const keys = Object.keys(chain);
  let key = keys[Math.floor(Math.random() * keys.length)];
  const words = key.split(" ");

  for (let i = 0; i < length; i++) {
    const options = chain[key];
    if (!options) break;
    const next = options[Math.floor(Math.random() * options.length)];
    words.push(next);
    key = words.slice(-order).join(" ");
  }
  return words.join(" ");
}

class LinkedList:
    class Node:
        def __init__(self, element):
            self.element = element
            self.next = None

    def __init__(self):
        self.length = 0
        self.head = None

    def is_empty(self):
        return self.length == 0

    def add(self, element):
        node = self.Node(element)
        if self.is_empty():
            self.head = node
        else:
            current_node = self.head
            while current_node.next is not None:
                current_node = current_node.next
            current_node.next = node
        self.length += 1

    def remove(self, element):
        previous_node = None
        current_node = self.head
        while current_node is not None and current_node.element != element:
            previous_node = current_node
            current_node = current_node.next
        if current_node is None:
            return
        elif previous_node is not None:
            previous_node.next = current_node.next
        else:
            self.head = current_node.next
        self.length -= 1

                return False
        return True

    # Recursive DFS function to place queens row by row
    def place_queen(board, row):
        if row == n:
            solutions.append(board.copy())  # Found valid solution
            return
        for col in range(n):
            if is_safe(board, row, col):
                board[row] = col
                place_queen(board, row + 1)
                # No need to unset board[row] because it will be overwritten in next iteration

    place_queen([0] * n, 0)
    return solutions

    let n = len(arr)
    let i = 0
    while i < n {
        let j = 0
        while j < n - i - 1 {
            if arr[j] > arr[j+1] {
